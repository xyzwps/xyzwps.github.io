Go 中并没有 `instanceof` 这种操作符。判断一个值是否是某种类型的，用强制类型转换就行：

```go
package main

type MyError struct {
	Message string
}

type Messaged interface {
	GetMessage() string
}

func (e *MyError) Error() string {
	return e.Message
}

func (e *MyError) GetMessage() string {
	return e.Message
}

func main() {
	var err error = &MyError{"xxxxx"}

	_, ok := err.(error)
	println(ok)  // true
	_, ok = err.(Messaged)
	println(ok)  // true
}
```
