题目96. 不同的二叉搜索树。

DP 解法

沿着第 95 题——不同的二叉搜索树 II的官方题解思路，很容易就可以写出下面的代码， 还是 DP 代码：

function numTrees(n: number): number {
    const dp: number[] = [0, 1, 2] // 这里强行指定了 n 比较小的情况的值
    for (let i = 3; i <= n; i++) {
        let r: number = 0;
        for (let t = 0; t <= i; t++) {
            if (t == 0) r += dp[i - 1]        // 没有左子树的情况
            else if (t == i) r += dp[i - 1]   // 没有右子树的情况
            else r += (dp[t] * dp[i - 1 - t]) // 有左右子树的情况
        }
        dp[i] = r
    }
    return dp[n]
};

上面的代码可以改成这样：

function numTrees(n: number): number {
    const dp: number[] = [1] // 注意，这里 dp[0] == 1
    for (let i = 1; i <= n; i++) {
        let r: number = 0;
        for (let t = 0; t < i; t++) {
            r += (dp[t] * dp[i - 1 - t])
        }
        dp[i] = r
    }
    return dp[n]
};
