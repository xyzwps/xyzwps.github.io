最小路径和

题目64. 最小路径和。

这题思路和不同路径这题一模一样，不解释了，直接上代码：

function minPathSum(grid: number[][]): number {
    const dp: number[][] = []
    const m = grid.length;
    const n = grid[0].length;
    for (let i = 0; i < m; i++) {
        dp[i] = []
        for (let j = 0; j < n; j++) {
            if (i == 0 && j == 0) {
                dp[i][j] = grid[i][j]
            } else if (i == 0) {
                dp[i][j] = grid[i][j] + dp[i][j - 1]
            } else if (j == 0) {
                dp[i][j] = grid[i][j] + dp[i - 1][j]
            } else {
                dp[i][j] = grid[i][j] + Math.min(dp[i - 1][j], dp[i][j - 1])
            }
        }
    }
    return dp[m - 1][n - 1]
};
